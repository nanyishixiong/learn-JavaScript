# 五、函数柯里化

函数柯里化：将一个多元函数转变为一元函数的过程。 每当函数被调用时，它仅仅接收一个参数并且返回带有一个参数的函数，直到传递完所有的参数。

# 利用偏函数，来实现柯里化

/**
 * 偏函数，预设一部分参数，等待下次参数传递
 * @param {function} fun
 * @returns
 */
function sub_curry(fun) {
  let slice = Array.prototype.slice
  // 去掉第一个参数，拿到剩下的参数
  let args = slice.call(arguments, 1)
  return function () {
    let newArgs = args.concat(slice.call(arguments))
    return fun.apply(this, newArgs)
  }
}

function curry(fn, length) {
  length = length || fn.length

  let slice = Array.prototype.slice

  return function () {
    if (arguments.length < length) {
    // 这一步，将fn函数放在数组第一位，apply函数会将数组展开一个个填进sub_curry函数的参数中
      let combined = [fn].concat(slice.call(arguments))
      return curry(sub_curry.apply(this, combined), length - arguments.length)
    } else {
      return fn.apply(this, arguments)
    }
  }
}

var fn = curry(function (a, b, c) {
  return [a, b, c];
})

fn("a", "b", "c") // ["a", "b", "c"]
fn("a", "b")("c") // ["a", "b", "c"]
fn("a")("b")("c") // ["a", "b", "c"]
fn("a")("b", "c") // ["a", "b", "c"]

实际上，这不仅是函数柯里化了，不仅可以预设一个参数，还可以一次预设多个参数，就像是偏函数的升级版，可以无限调用，直到参数满足函数调用的要求为止。

# 简单版本的实现

function curry(fn, args) {
  var length = fn.length;
  args = args || []

  return function () {
    var _args = args.slice();

    _args = _args.concat([].slice.call(arguments))

    if (_args.length < length) {
      return curry.call(this, fn, _args)
    } else {
      return fn.apply(this, _args)
    }

  }
}

# 复杂版curry函数——占位符

// 第三版
function curry(fn, args, holes) {
  length = fn.length;

  args = args || [];

  holes = holes || [];

  return function () {

    var _args = args.slice(0),
      _holes = holes.slice(0),
      argsLen = args.length,
      holesLen = holes.length,
      arg, i, index = 0;

    for (i = 0; i < arguments.length; i++) {
      console.log(holesLen);
      arg = arguments[i];
      // 处理类似 fn(1, _, _, 4)(_, 3) 这种情况，index 需要指向 holes 正确的下标
      if (arg === _ && holesLen) {
        index++
        if (index > holesLen) {
          _args.push(arg);
          _holes.push(argsLen - 1 + index - holesLen)
        }
      }
      // 处理类似 fn(1)(_) 这种情况
      else if (arg === _) {
        _args.push(arg);
        _holes.push(argsLen + i);
      }
      // 处理类似 fn(_, 2)(1) 这种情况
      else if (holesLen) {
        // fn(_, 2)(_, 3)
        if (index >= holesLen) {
          _args.push(arg);
        }
        // fn(_, 2)(1) 用参数 1 替换占位符
        else {
          _args.splice(_holes[index], 1, arg);
          _holes.splice(index, 1)
        }
      }
      else {
        _args.push(arg);
      }

    }
    if (_holes.length || _args.length < length) {
      return curry.call(this, fn, _args, _holes);
    }
    else {
      return fn.apply(this, _args);
    }
  }
}

var _ = {};

var fn = curry(function (a, b, c, d, e) {
  console.log([a, b, c, d, e]);
});

// 验证 输出全部都是 [1, 2, 3, 4, 5]
fn(1, 2, 3, 4, 5);
fn(_, 2, 3, 4, 5)(1);
fn(1, _, 3, 4, 5)(2);
fn(1, _, 3)(_, 4)(2)(5);
fn(1, _, _, 4)(_, 3)(2)(5);
fn(_, 2)(_, _, 4)(1)(3)(5)
