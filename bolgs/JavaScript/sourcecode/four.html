    # 四、偏函数

    偏函数："部分地"应用一个函数，即预设原始函数的部分参数来创建一个新的函数。

    Curried functions are automatically partially applied. 柯里化就是自动实现的偏函数

    # 实现偏函数

    // 创建偏函数，固定一些参数
function partical(fn, ...args) {
  // 返回一个带有剩余参数的函数
  return function (...moreArgs) {
    // 调用原始函数
    return fn(...args, ...moreArgs)
  }
}

const add3 = (a, b, c) => a + b + c // (c) => 2 + 3 + c

// 部分地将`2`和`3`应用于`add3`，得到一个只有一个参数的函数
const fivePlus = partical(add3, 2, 3)
fivePlus(4)  // 9

    我们可以利用ES6箭头函数让他变得更加简洁

    const partical = (fn, ...args) =>
  (...moreArgs) =>
    fn(...args, ...moreArgs)

    # bind方法实现偏函数

    当然，我们也可以用Function.prototype.bind来部分应用一个函数

    // 也可以用bind方法实现偏函数
const addMore = add3.bind(null, 2, 3)
addMore(1) // 6

    # 偏函数 this 指向

    function add(a, b) {
  return a + b + this.value;
}

// var addOne = add.bind(null, 1);
var addOne = partical(add, 1);

var value = 1;
var obj = {
  value: 2,
  addOne: addOne
}

obj.addOne(2) // NaN

    这里不管是partical，还是用bind方法，输出都是NaN

    原因： add 方法无人调用this指向window，严格模式指向 undefined ，这里我在

    node环境下运行，所以指向undefine; 如果是在浏览器环境下，那就是输出4

    # 升级版——不改变this指向

    // 创建偏函数，固定一些参数
function partical(fn, ...args) {
  // 返回一个带有剩余参数的函数
  return function (...moreArgs) {
    // 调用原始函数
    return fn.apply(this, [...args, ...moreArgs])
  }
}

    用apply方法将函数指回调用函数的对象，这回再尝试刚刚的例子输出就是5了

    # 假如不可使用ES6，扩展运算符...

    function partical(fn) {
  // 取出除了第一个参数，剩下的参数
  var args = [].slice.call(arguments, 1)
  return function () {
    // jian
    var moreArgs = [].slice.call(arguments)
    var newArgs = args.concat(moreArgs)
    return fn.apply(this, newArgs)
  }
}

    # Pro版——可以使用占位符

    const add3 = (a, b, c) => a + b + c

    现在我想预设b参数或者c参数，那我们上面的实现就没法达到目的

    var _ = {};
function partical(fn) {
  var args = [].slice.call(arguments, 1)
  return function () {
    var position = 0, len = args.length;
    for (var i = 0; i < len; i++) {
      // 如果为空就用 arguments 里面的参数代替
      args[i] = args[i] === _ ? arguments[position++] : args[i]
    }
    while (position < arguments.length) args.push(arguments[position++])
    return fn.apply(this, args)
  }
}


function add(a, b) {
  return a + b + this.value;
}

var addOne = partical(add, _, 2);

var value = 1;
var obj = {
  value: 2,
  addOne: addOne
}
console.log(obj.addOne(2)); // 6

    Nice，完美实现！
